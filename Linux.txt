pwd - где мы находимся
ls -la - что находится в папке
mkdir папка_1 папка_2  - создавать папки
touch qq.txt - создать файл
cat >> qq.txt можно записывать в файл уже созданный
cat qq.txt просмотреть содержимое файла
cat > qq_1.txt создать файл и начать записывать в него
cd название папки - зайти в папку
сd .. перейти на уровень више
history - посмотреть последние ввенные команды
vim q1.txt тедактор файлов
:wq выйти и сохранить из вим
ls -la имя папки - узнать что в папке не заходя в нее
mv qa_1/q1.txt(от куда)  qa_2/q11.txt(куда) переместить файл
rm название файла - удалить файл
rm -r - удалить файл
mv q11.txt q99.txt переименовать файл
rmdir qa_4 - удалить папку (не удаляет не пустую папку)
